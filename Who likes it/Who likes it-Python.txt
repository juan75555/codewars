def likes(names):
    s = ''
    if(len(names)<1):
        return 'no one likes this'
    else:
        if(len(names)==1):
            s+= names[0]+ ' likes this'
        elif(len(names)==2):
            s += names[0]+' and '+names[1]+ ' like this'
        elif(len(names)==3):
            s += names[0]+', '+names[1] + ' and ' +names[2]+ ' like this'
        else:
            s = names[0]+', '+names[1] + ' and {} others like this'.format(len(names)-2)
    return s
